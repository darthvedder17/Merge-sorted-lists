#Function for finding longest common prefix of a string

def index(a):
    n=len(a)
    min = len(a)

    for i in range(1, n):
        if len(a[i]) < min:
            min = len(a[i])

    newmin = min
    result = ''
    for j in range(newmin):
        current = a[0][j]
        for k in range(1, n):
            if a[k][j] != current:
                return result
        result = result + current
    return result

a=['abcde','abcdegf','abc','ab']
obj=index(a)
print(obj)
